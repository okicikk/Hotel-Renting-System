pipeline {
    agent any
    environment {
        DOTNET_CLI_HOME = "${env.TEMP}"  // Use Windows temp directory
    }
    stages {
        stage('Restore') {
            steps {
                bat 'dotnet restore'
            }
        }
        stage('Build') {
            steps {
                bat 'dotnet build --no-restore --configuration Release'
            }
        }
        stage('Test') {
            steps {
                bat 'dotnet test --no-build --no-restore --configuration Release'
            }
        }
    }
    post {
        always {
            cleanWs()
        }
    }
}
